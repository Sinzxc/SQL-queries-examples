Пример insert

insert into Product values('fsdfsdf',123,'Майкоп')

INSERT INTO Product: Эта часть запроса указывает, что вы хотите вставить данные в таблицу "Product"
VALUES ('fsdfsdf', 123, 'Майкоп'): Здесь предоставляются значения, которые вы хотите вставить в соответствующие столбцы таблицы "Product". Значения предоставляются в том же порядке, что и столбцы в таблице.
Первое значение 'fsdfsdf' соответствует первому столбцу в таблице "Product".
Второе значение 123 соответствует второму столбцу.
Третье значение 'Майкоп' соответствует третьему столбцу.
Предполагая, что у таблицы "Product" три столбца (в порядке: column1, column2, column3), предоставленный запрос INSERT вставит новую строку в таблицу "Product" со значениями 'fsdfsdf', 123 и 'Майкоп' для соответствующих столбцов.

Пример delete

delete from Product where Product.Name='fsdfsdf';

DELETE FROM Product: Эта часть запроса указывает, что вы хотите удалить данные из таблицы "Product".
WHERE Product.Name = 'fsdfsdf': Это условие определяет, какие строки будут удалены. В данном случае, удаляются только те строки, где значение в столбце "Name" равно 'fsdfsdf'.
Таким образом, этот запрос удалит все строки из таблицы "Product", где значение в столбце "Name" соответствует 'fsdfsdf'. Убедитесь, что такие строки действительно существуют в вашей таблице перед выполнением этой операции, так как удаление данных необратимо.

Пример простого select

select Postavka.Id_provide
from Postavka
where count=100

SELECT Id_provide: Эта часть запроса указывает, какие столбцы будут выбраны. В данном случае, выбирается столбец "Id_provide" из таблицы "Postavka".
FROM Postavka: Это указывает, из какой таблицы будут выбраны данные. В данном случае, данные выбираются из таблицы "Postavka".
WHERE Count = 100: Это условие фильтрации данных. Здесь предполагается, что "Count" - это столбец в таблице "Postavka", и выбираются только те строки, где значение в этом столбце равно 100.

Пример задачи1 (group, вложенные запросы)

select Id_provide from Postavka where count>(select Max(count) from Postavka where Id_provide=1) Group by Id_provide

Получить номера поставщиков, которые поставляют, по крайней мере, один товар, в количестве большем, чем максимальное количество поставленных когда-либо товаров поставщиком с номером S1.
